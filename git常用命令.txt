git init #进入本地目录

git --version #查看git版本

git config --global user.name "你的用户名" #配置用户名
git config --global user.email "你的邮箱" #配置邮箱

git clone https://github.com/youname/repo #克隆一个仓库或者仓库下的文件

git add 文件名 #添加文件

git rm [-rf] 文件名  #删除文件
git reset --hard HEAD #撤销和回退

git commit -m "你的备注" #提交并备注

git push [-u] origin master #推送到master分支
